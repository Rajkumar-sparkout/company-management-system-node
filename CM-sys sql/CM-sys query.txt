CREATE TABLE IF NOT EXISTS state(
  state_id int NOT NULL auto_increment,
  name varchar(50) not null,
  PRIMARY KEY (state_id)
);

CREATE TABLE IF NOT EXISTS district(
  district_id int NOT NULL auto_increment,
  district varchar(50) not null ,
  state int not null, 
  PRIMARY KEY (district_id),
  foreign key(state) references state(state_id)
);

CREATE TABLE IF NOT EXISTS industry(
  id int NOT NULL auto_increment,
  industry varchar(50) not null,
  PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS roles( 
  id int NOT NULL auto_increment,
  name varchar(50) NOT NULL,
  PRIMARY KEY (id)
);
select * from roles;

CREATE TABLE IF NOT EXISTS membership( 
  id int NOT NULL AUTO_INCREMENT,
  type varchar(50) NOT NULL,
  PRIMARY KEY (`id`)
);
select * from membership;

CREATE TABLE IF NOT EXISTS user_otp( 
  id bigint NOT NULL auto_increment,
  mobile_number varchar(50) NOT NULL UNIQUE KEY,
  user_otp varchar(50) NOT NULL,
  created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  last_modified_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id)
);
select * from user_otp;

CREATE TABLE IF NOT EXISTS business_user( 
  business_user_id bigint NOT NULL auto_increment,
  company_name varchar(50) NOT NULL,
  owner varchar(50) NOT NULL,
  email varchar(50) NOT NULL UNIQUE KEY,
  mobile_number varchar(50) NOT NULL UNIQUE KEY,
  industry varchar(50),
  website varchar(50),
  company_description varchar(100),
  tag varchar(50),
  PRIMARY KEY (business_user_id)
);
select * from business_user;

CREATE TABLE IF NOT EXISTS business_user_address( 
  id bigint NOT NULL auto_increment,
  address varchar(50),
  city varchar(50),
  district varchar(50),
  state varchar(50),
  zipcode varchar(50),
  business_user_id bigint not null,
  PRIMARY KEY (id),
  foreign key(business_user_id) references business_user(business_user_id)
);
select * from business_user_address;

CREATE TABLE IF NOT EXISTS user_account( 
  id bigint NOT NULL auto_increment,
  user_name varchar(20) NOT NULL UNIQUE KEY,
  email varchar(50) NOT NULL UNIQUE KEY,
  password varchar(120) NOT NULL, 
  active int not null,
  business_user_id bigint,
  user_id bigint,
  password_expire_date DATE not null,
  password_expired int not null,
  created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  last_modified_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  foreign key(business_user_id) references business_user(business_user_id),
  foreign key(user_id) references users(user_id)
);
select * from user_account;

SET FOREIGN_KEY_CHECKS=0;
SET GLOBAL FOREIGN_KEY_CHECKS=0;

CREATE TABLE IF NOT EXISTS user_roles( 
  business_user_id bigint,
  user_id bigint,
  role_id int NOT NULL,
  foreign key(business_user_id) references business_user(business_user_id),
  foreign key(user_id) references users(user_id),
  foreign key(role_id) references roles(id)
);
select * from user_roles;

CREATE TABLE IF NOT EXISTS user_membership( 
  id bigint NOT NULL auto_increment,
  membership_type int NOT NULL,
  active int NOT NULL,
  business_user_id bigint NOT NULL,
  created_date timestamp  DEFAULT CURRENT_TIMESTAMP,
  expired_date timestamp  DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  FOREIGN KEY (business_user_id ) REFERENCES business_user  (business_user_id ),
  FOREIGN KEY (membership_type) REFERENCES membership (id)
);
select * from user_membership;

CREATE TABLE IF NOT EXISTS users( 
  user_id bigint NOT NULL auto_increment,
  first_name varchar(50) NOT NULL,
  last_name varchar(50),
  email varchar(50) NOT NULL UNIQUE KEY,
  mobile_number varchar(50) NOT NULL UNIQUE KEY,
  tag varchar(50),
  created_by int,
  PRIMARY KEY (user_id)
);
select * from users;

CREATE TABLE IF NOT EXISTS user_address( 
  id bigint NOT NULL auto_increment,
  address varchar(50),
  city varchar(50),
  district varchar(50),
  state varchar(50),
  zipcode varchar(50),
  user_id bigint not null,
  PRIMARY KEY (id),
  foreign key(user_id) references users(user_id)
);
select * from user_address;

CREATE TABLE IF NOT EXISTS customer( 
  customer_id bigint NOT NULL auto_increment,
  customer_number varchar(50) NOT NULL,
  customer_name varchar(50) NOT NULL,
  email varchar(50) NOT NULL UNIQUE KEY,
  mobile_number varchar(50) NOT NULL UNIQUE KEY,
  whatsapp_number varchar(50),
  PRIMARY KEY (customer_id)
);
select * from customer;

CREATE TABLE IF NOT EXISTS customer_address( 
  id bigint NOT NULL auto_increment,
  address varchar(50),
  city varchar(50),
  district varchar(50),
  state varchar(50),
  zipcode varchar(50),
  customer_id bigint not null,
  PRIMARY KEY (id),
  foreign key(customer_id) references customer(customer_id)
);
